# -*- coding: utf-8 -*-
from __future__ import unicode_literals

from django.shortcuts import render
from django.http import HttpResponseRedirect, Http404

from .models import Plan
from django.contrib.auth.decorators import login_required


# Create your views here.
def index(request):
	return render(request, 'meal_planners/index.html')
	
def plans(request):
	plans = Plan.objects.filter(owner=request.user).order_by('date_added')
	context = {'plans':plans}
	return render(request,'meal_planners/plans.html',context)

	
	
def plan(request,plan_id):
	plan = Plan.objects.get(id=plan_id)
	if plan.owner != request.user:
		raise Http404
	details = plan.entry_set.order_by('-date_added')	
	context = {'plan':plan,'details':details}
	return render(request, 'meal_planners/plan.html',context)


@login_required
def plans(request):
	plans = Plan.objects.filter(owner=request.user).order_by('date_added')
	context = {'plans':plans}
	return render(request,'meal_planners/plans.html',context)
